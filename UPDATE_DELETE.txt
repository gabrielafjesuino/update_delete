/* 1 - Atualize o peso do paciente com ID 1 para 8.5 kg */

UPDATE PACIENTE 
SET PESO = 8.5
WHERE ID_PACIENTE = 1

/* 2 - Atualize o estoque do medicamento com ID 3 para 50 unidades */

UPDATE MEDICAMENTOS 
SET ESTOQUE = 50
WHERE ID_MEDICAMENTO = 3

/* 3 - Atualize a data de nascimento do tutor com ID 2 para '15/05/2001' */

UPDATE TUTOR 
SET DATA_NASCIMENTO = '2001-05-15'
WHERE ID_TUTOR = 2

/* 4 - Atualize o motivo da consulta com ID 5 para 'Dor de ouvido severa' */

UPDATE AGENDAMENTO 
SET MOTIVO = 'Dor de ouvido severa'
WHERE ID_AGENDAMENTO = 5

/* 5 - Atualize a descrição do medicamento com ID 6 para 'Anti-inflamatório para articulações' */

UPDATE MEDICAMENTOS 
SET DESCRICAO = 'Anti-inflamatório para articulações'
WHERE ID_MEDICAMENTO = 6

/* 6 - Atualize o nome do tutor com ID 3 para 'Maria Oliveira' */

UPDATE TUTOR 
SET NOME = 'Maria Oliveira'
WHERE ID_TUTOR = 3

/* 7 - Atualize a posologia do medicamento com ID 2 para 'Aplicar no olho afetado a cada 4 horas' */

UPDATE MEDICAMENTOS 
SET DESCRICAO = 'Aplicar no olho afetado a cada 4 horas'
WHERE ID_MEDICAMENTO = 2

/* 8 - Atualize o endereço do tutor com ID 4 para 'Rua dos Girassóis, 123' */

UPDATE TUTOR 
SET ENDERECO = 'Rua dos Girassóis, 123'
WHERE ID_TUTOR = 4

/* 9 - Delete a consulta com ID 7 */

DELETE FROM AGENDAMENTO 
WHERE ID_AGENDAMENTO = 7

/* 10 - Delete o medicamento com ID 8 */

DELETE FROM MEDICAMENTOS 
WHERE ID_MEDICAMENTO = 8

/* 11 - Delete os animais do tutor que tem o ID 5 */

DELETE 
  FROM RECEITA r 
 WHERE r.ID_AGENDAMENTO IN (SELECT a.ID_AGENDAMENTO 
                              FROM AGENDAMENTO a
                             WHERE a.ID_PACIENTE IN (SELECT p.ID_PACIENTE 
                                                       FROM PACIENTE p 
                                                      WHERE p.ID_TUTOR = 5))
                                                      
DELETE 
  FROM AGENDAMENTO a
 WHERE a.ID_PACIENTE IN (SELECT p.ID_PACIENTE 
                           FROM PACIENTE p 
                          WHERE p.ID_TUTOR = 5)

DELETE 
  FROM PACIENTE p
 WHERE p.ID_TUTOR = 5

/* 12 - Delete todas as receitas e seus itens que não estão relacionadas a nenhuma consulta */

DELETE 
  FROM ITENSRECEITA i 
 WHERE i.ID_RECEITA IN (SELECT r.ID_RECEITA
                          FROM RECEITA r 
                         WHERE r.ID_AGENDAMENTO NOT IN (SELECT a.ID_AGENDAMENTO 
                                                           FROM AGENDAMENTO a 
                                                           WHERE a.ID_AGENDAMENTO = r.ID_AGENDAMENTO)
)

DELETE 
  FROM RECEITA r 
 WHERE r.ID_AGENDAMENTO NOT IN (SELECT a.ID_AGENDAMENTO 
                                  FROM AGENDAMENTO a 
                                 WHERE a.ID_AGENDAMENTO = r.ID_AGENDAMENTO)

/* 13 - Delete todos os tutores que nasceram antes de 1987 */

DELETE FROM TUTOR 
WHERE DATA_NASCIMENTO < '1987-01-01'

/* 14 - Delete todos os itens da receita, das receitas que tem o intervalo de ID 7 ao 9 */

DELETE FROM ITENSRECEITA 
WHERE ID_RECEITA BETWEEN 7 AND 9
